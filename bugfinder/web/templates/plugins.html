Plugin SDK

Create plugins using the BugFinder Plugin SDK:

from bugfinder.plugins.sdk import (
    PluginBase,
    pre_scan, post_scan,
    on_finding, on_report,
)

class MyPlugin(PluginBase):
    name = "my-plugin"
    version = "1.0.0"
    description = "My custom plugin"

    def register(self, registry):
        registry.register_plugin(self.name, self)

@on_finding
def log_finding(finding):
    print(f"Found: {finding.title}")

Save as ~/.config/bugfinder/plugins/<name>.py
